# Complete the howManyGames function below.
def howManyGames(p, d, m, s):
    ans = 0
    while s >= m and p <= s:
        if p <= m:
            ans += 1
            s -= m
        else:
            ans += 1
            s -= p
            p -= d
    return(ans)
            

if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    pdms = input().split()

    p = int(pdms[0])

    d = int(pdms[1])

    m = int(pdms[2])

    s = int(pdms[3])

    answer = howManyGames(p, d, m, s)

    fptr.write(str(answer) + '\n')

    fptr.close()
